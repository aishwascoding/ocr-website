# Start from official Python image
FROM python:3.10-slim

# Install system dependencies
RUN apt-get update && \
    apt-get install -y --no-install-recommends \
        tesseract \
        tesseract-data-eng \
        tesseract-data-hin \
        tesseract-data-fra \
        tesseract-data-spa \
        tesseract-data-chi_sim \
        tesseract-data-ara \
        tesseract-data-deu \
        tesseract-data-jpn \
        tesseract-data-kor \
        tesseract-data-urd \
        libgl1 \
        libsm6 \
        ffmpeg \
        zlib1g \
        imagemagick \
        gunicorn \
        python3-flask \
        python3-pillow && \
    apt-get clean && \
    rm -rf /var/lib/apt/lists/*

# Set working directory
WORKDIR /app

# Copy only necessary files
COPY app.py .
COPY templates templates/
COPY static static/

# Confirm Tesseract is installed in expected location
RUN if [ ! -f /usr/bin/tesseract ]; then echo "‚ùå ERROR: Tesseract NOT found!" && exit 1; fi

# Expose port and run with Gunicorn
EXPOSE 8000
CMD ["gunicorn", "--bind", "0.0.0.0:8000", "app:app"]
